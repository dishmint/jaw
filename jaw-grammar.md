# Grammar

```
SOURCE        => REPEAT( VARIABLE | FUNCTION | COMMENT | TEXT )

VARIABLE      => [ IDENTIFIER ] — TEXT DECORATOR*
VARIABLE_REF  => [ IDENTIFIER ]
INLINE_ASSIGN => [ IDENTIFIER ] : TEXT = VALUE
INLINE_ASSIGN => [ IDENTIFIER ] : TEXT

FUNCTION_DEF  => / IDENTIFIER DECORATOR* FUNCTION_ARGS? NEWLINE CODEBLOCK
FUNCTION_REF  => / IDENTIFIER
FUNCTION_CALL => / IDENTIFIER [ CALL_ARGS ]
FUNCTION_ARGS => ARGLIST
FUNCTION_ARGS => NEWLINE ARGLIST
ARGLIST       => INLINE_ASSIGN ( , INLINE_ASSIGN )*
CALL_ARGS     => EXPRESSION ( , EXPRESSION )*

CODEBLOCK     => REPEAT( INLINE_ASSIGN | STEP | COMPLEX_COND | LOOP | PARALLEL | COMMENT | LOG | RETURN )

STEP          => [ NUMBER ] — EXPRESSION DECORATOR*
EXPRESSION    => CONDITIONAL | CODE
CONDITIONAL   => CODE ? FUNCTION_CALL ( | CODE ? FUNCTION_CALL )* ( | FUNCTION_CALL )?
CODE          => TEXT

LOOP          => [~] — LOOP_EXPR NEWLINE CODEBLOCK
LOOP_EXPR     => CONDITION | ITERATION
CONDITION     => CODE
ITERATION     => VARIABLE_REF in VARIABLE_REF
ITERATION     => ( VARIABLE_REF ( , VARIABLE_REF )* ) in VARIABLE_REF

PARALLEL      => [&] NEWLINE CODEBLOCK

COMPLEX_COND  => [ NUMBER ] — CODE ? NEWLINE TRUE_BLOCK FALSE_BLOCK
TRUE_BLOCK    => INDENT [+] — CODE NEWLINE
FALSE_BLOCK   => INDENT [-] — CODE NEWLINE

RETURN        => [>] CODE DECORATOR*

LOG           => [!] — LOG_TITLE? DECORATOR? LOG_CONTENT
LOG_TITLE     => TEXT : NEWLINE  /* text ending in colon, italicized */
LOG_CONTENT   => TEXT CONTINUATION*
LOG_CONTENT   => NEWLINE INDENT TEXT CONTINUATION*  /* multi-line, indentation-based */

CODE_COMMENT  => [^] TEXT CONTINUATION*
GEN_COMMENT   => [*] TEXT CONTINUATION*
COMMENT       => CODE_COMMENT | GEN_COMMENT
CONTINUATION  => NEWLINE TEXT  /* plain text not starting with a JAW marker */

DECORATOR     => # IDENTIFIER
DECORATOR     => # IDENTIFIER : VALUE

ACCESS        => VARIABLE_REF @ VARIABLE_REF

IDENTIFIER    => A-Z a-z 0-9 _
NUMBER        => 0-9+
VALUE         => PLAINTEXT
TEXT          => PLAINTEXT
```
